# Production Environment Configuration for Forensic AI Log Analyzer
# Copy this content to your .env file and fill in the values

# =============================================================================
# REQUIRED SETTINGS (Must be configured for production)
# =============================================================================

# LLM Provider Selection (IBM Granite only)
LLM_PROVIDER=ibm_granite

# IBM Watsonx.ai Configuration (Required)
IBM_API_KEY=your_ibm_api_key_here
IBM_PROJECT_ID=your_ibm_project_id_here
IBM_URL=https://us-south.ml.cloud.ibm.com

# AWS S3 Configuration (Optional - for file storage only)
AWS_ACCESS_KEY_ID=your_aws_access_key_here
AWS_SECRET_ACCESS_KEY=your_aws_secret_key_here
AWS_DEFAULT_REGION=ap-south-1
AWS_S3_BUCKET=your_s3_bucket_name

# Security Settings (Required for production)
SECRET_KEY=your_32_character_secret_key_here
DEBUG_MODE=false
ALLOWED_HOSTS=localhost,127.0.0.1,your-domain.com
CORS_ORIGINS=https://your-domain.com,http://localhost:8501

# =============================================================================
# OPTIONAL SETTINGS (Configure as needed)
# =============================================================================

# VirusTotal Configuration (Optional - for IOC enrichment)
VT_API_KEY=your_virustotal_api_key_here

# ChromaDB Configuration
CHROMA_PERSIST_DIR=./chroma_db
CHROMA_COLLECTION_NAME=security_logs

# Application Settings
LOG_LEVEL=INFO
MAX_LOG_ENTRIES=10000

# Rate Limiting (Optional)
RATE_LIMIT_REQUESTS=100
RATE_LIMIT_WINDOW=3600

# =============================================================================
# PRODUCTION NOTES
# =============================================================================
# 1. IBM Granite is the ONLY LLM provider supported
# 2. Ensure your IBM Watsonx.ai account has access to Granite models
# 3. Set DEBUG_MODE=false in production
# 4. Use a strong SECRET_KEY (32+ characters)
# 5. Configure proper ALLOWED_HOSTS and CORS_ORIGINS for your domain
